# Plamp CLI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a `plamp` CLI that manages config, timers, and pictures through the existing `plamp-web` HTTP API, with JSON-first output and explicit binary streaming for picture downloads.

**Architecture:** Add a dedicated `plamp_cli/` package that stays separate from `plamp_web/` and acts as a thin HTTP client over the current API. Keep the command surface compact, centralize target resolution and JSON/binary I/O helpers, and ship a strong `plamp_cli/README.md` contract so agents can use the CLI directly or through a later thin skill wrapper.

**Tech Stack:** Python 3.11, `argparse`, stdlib `json`, stdlib `urllib`, stdlib `unittest`, existing `uv` workflow, existing FastAPI API endpoints.

```python
from __future__ import annotations

import json
import os
from dataclasses import dataclass
from typing import Any
from urllib.error import HTTPError, URLError
from urllib.parse import urlencode
from urllib.request import Request, urlopen


class ApiError(RuntimeError):
    def __init__(self, status: int, detail: str):
        super().__init__(f"API {status}: {detail}")
        self.status = status
        self.detail = detail


class NetworkError(RuntimeError):
    pass


def build_base_url(host: str | None, port: int | None, base_url: str | None) -> str:
    if base_url:
        return base_url.rstrip("/")
    env_base_url = os.environ.get("PLAMP_BASE_URL", "").strip()
    if env_base_url:
        return env_base_url.rstrip("/")
    resolved_host = host or os.environ.get("PLAMP_HOST") or "127.0.0.1"
    resolved_port = port or int(os.environ.get("PLAMP_PORT") or "8000")
    return f"http://{resolved_host}:{resolved_port}"


def request_json(method: str, base_url: str, path: str, payload: Any | None = None, query: dict[str, Any] | None = None) -> Any:
    query_string = f"?{urlencode(query)}" if query else ""
    request = Request(f"{base_url}{path}{query_string}", method=method)
    data = None
    if payload is not None:
        data = json.dumps(payload).encode("utf-8")
        request.add_header("Content-Type", "application/json")
    try:
        with urlopen(request, data=data) as response:
            return json.loads(response.read().decode("utf-8"))
    except HTTPError as exc:
        detail = exc.read().decode("utf-8").strip() or exc.reason
        raise ApiError(exc.code, detail) from exc
    except URLError as exc:
        raise NetworkError(str(exc.reason)) from exc
```

from __future__ import annotations

import json
import sys
from pathlib import Path
from typing import Any, TextIO


class InputError(RuntimeError):
    pass


def load_json_input(value: str, stdin: TextIO | None = None) -> Any:
    source = stdin or sys.stdin
    if value == "-":
        raw = source.read()
    elif value.startswith("@"):
        raw = Path(value[1:]).read_text(encoding="utf-8")
    else:
        raise InputError("expected @file.json or - for stdin")
    try:
        return json.loads(raw)
    except json.JSONDecodeError as exc:
        raise InputError(f"invalid JSON input: {exc.msg}") from exc


def format_json_output(data: Any, pretty: bool = False) -> str:
    if pretty:
        return json.dumps(data, indent=2, sort_keys=True) + "\n"
    return json.dumps(data, sort_keys=True) + "\n"
```

from plamp_cli.http import build_base_url, request_json
from plamp_cli.io import format_json_output, load_json_input


def build_parser() -> argparse.ArgumentParser:
    parser = argparse.ArgumentParser(prog="plamp")
    parser.add_argument("--host")
    parser.add_argument("--port", type=int)
    parser.add_argument("--base-url")
    parser.add_argument("--pretty", action="store_true")
    parser.add_argument("--table", action="store_true")
    subparsers = parser.add_subparsers(dest="area", required=True)

    config = subparsers.add_parser("config")
    config_subparsers = config.add_subparsers(dest="section", required=True)
    config_root = config_subparsers.add_parser("get")
    config_root.set_defaults(handler=handle_config_get, section_name=None)
    config_put = config_subparsers.add_parser("set")
    config_put.add_argument("payload")
    config_put.set_defaults(handler=handle_config_set, section_name=None)
    for name in ("controllers", "devices", "cameras"):
        section = config_subparsers.add_parser(name)
        section_actions = section.add_subparsers(dest="action", required=True)
        section_get = section_actions.add_parser("get")
        section_get.set_defaults(handler=handle_config_get, section_name=name)
        section_set = section_actions.add_parser("set")
        section_set.add_argument("payload")
        section_set.set_defaults(handler=handle_config_set, section_name=name)
    return parser


def handle_config_get(args: argparse.Namespace, base_url: str) -> object:
    response = request_json("GET", base_url, "/api/config")
    if args.section_name is None:
        return response
    return response["config"][args.section_name]


def handle_config_set(args: argparse.Namespace, base_url: str) -> object:
    payload = load_json_input(args.payload)
    path = "/api/config" if args.section_name is None else f"/api/config/{args.section_name}"
    return request_json("PUT", base_url, path, payload)
```

def main(argv: Sequence[str], stdout: TextIO | None = None, stderr: TextIO | None = None, stdout_buffer: Any | None = None) -> int:
    stdout = stdout or sys.stdout
    stderr = stderr or sys.stderr
    stdout_buffer = stdout_buffer or sys.stdout.buffer
    parser = build_parser()
    try:
        args = parser.parse_args(list(argv))
        base_url = build_base_url(args.host, args.port, args.base_url)
        result = args.handler(args, base_url)
        if isinstance(result, bytes):
            write_binary_output(result, getattr(args, "out", None), stdout_buffer)
        else:
            stdout.write(format_json_output(result, pretty=args.pretty))
        return 0
    except ApiError as exc:
        stderr.write(f"{exc}\n")
        return 3
    except NetworkError as exc:
        stderr.write(f"{exc}\n")
        return 4
    except InputError as exc:
        stderr.write(f"{exc}\n")
        return 5
    except ValueError as exc:
        stderr.write(f"{exc}\n")
        return 2
    except SystemExit as exc:
        return int(exc.code)
```

# Plamp CLI

`plamp` is a JSON-first command-line client for `plamp-web`.

## Defaults

- base URL: `http://127.0.0.1:8000`
- stdout: JSON
- stderr: diagnostics only

## Remote Use

```bash
plamp --host pi.local timers get pump_lights
ssh pi.local plamp config get
ssh pi.local plamp pics get grow:latest --stdout > latest.jpg
```

## JSON Input

```bash
plamp config set @config.json
cat state.json | plamp timers set pump_lights -
```

## Pictures

```bash
plamp pics list --source grow
plamp pics take --camera-id rpicam_cam0
plamp pics get grow:latest --out latest.jpg
plamp pics get grow:latest --stdout > latest.jpg
```

## Exit Codes

- `0` success
- `2` usage error
- `3` API error
- `4` network error
- `5` local input error
